function play(note) { if (!ctx || !buffer) return; const source = ctx.createBufferSource(); const gain = ctx.createGain(); source.buffer = buffer; // 🎹 12-TET (pianino) const semitone = note - 60; source.playbackRate.value = Math.pow(2, semitone / 12); // 🔔 DŁUŻSZY CARILLON (bardziej „wieża dzwonów”) gain.gain.setValueAtTime(1.0, ctx.currentTime); gain.gain.exponentialRampToValueAtTime(0.0001, ctx.currentTime + 3.8); source.connect(gain); gain.connect(ctx.destination); source.start(); }
